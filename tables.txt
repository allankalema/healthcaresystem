from datetime import datetime
from django.utils import timezone
from antenatal.models import AntenatalCard  # Adjust the import based on your app structure

# Fetch the AntenatalCard instance with id=1
antenatal_card = AntenatalCard.objects.get(id=1)

# Convert naive datetime to an aware datetime
next_visit_date = timezone.make_aware(datetime(2025, 3, 8, 0, 0))

# Update the next_visit field
antenatal_card.next_visit = next_visit_date

# Save the changes to the database
antenatal_card.save()

print(f"Updated next visit date: {antenatal_card.next_visit}")
